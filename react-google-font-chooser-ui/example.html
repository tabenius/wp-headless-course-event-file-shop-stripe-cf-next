react-google-font-chooser-ui

This example shows how to mount GoogleFontChooser in a React host app.

Usage

Use this snippet inside your React/Next component file. 

import { GoogleFontChooser } from "./src";

export default function FontChooserExample() {
  return (
    <GoogleFontChooser
      initialFamily="Inter"
      initialPreviewText="Typography for RAGBAZ storefront"
      onApply={(selection) => {
        console.log("Applied selection:", selection);
        // Persist to your style settings here.
      }}
    />
  );
}
